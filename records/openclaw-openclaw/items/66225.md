Keep this PR open. Current main still has the narrow final-tag regexes in both affected paths, so the self-closing `<final/>` leak described by the PR and linked #65867 is not implemented on main. The PR is a plausible focused fix, but its proposed streaming scanner still needs maintainer review for the malformed-attribute strict-mode leak noted in review comments.

Keep this PR open for maintainer review. After that lands, #65867 can be closed as implemented and overlapping open PRs can be deduplicated.

function stripWithPrRegex(processed) {
  const FINAL_TAG_SCAN_RE = /<\s*(\/?)\s*final\b[^<>]*>/gi;
  let result = '';
  let lastFinalIndex = 0;
  let inFinal = false;
  let everInFinal = false;
  for (const match of processed.matchAll(FINAL_TAG_SCAN_RE)) {
    const idx = match.index ?? 0;
    const isClose = match[1] === '/';
    const isSelfClosing = !isClose && /\/\s*>$/.test(match[0]);
    if (isSelfClosing) {
      if (inFinal) {
        result += processed.slice(lastFinalIndex, idx);
        inFinal = false;
      }
      lastFinalIndex = idx + match[0].length;
    } else if (!inFinal && !isClose) {
      inFinal = true;
      everInFinal = true;
      lastFinalIndex = idx + match[0].length;
    } else if (inFinal && isClose) {
      result += processed.slice(lastFinalIndex, idx);
      inFinal = false;
      lastFinalIndex = idx + match[0].length;
    }
  }
  if (inFinal) result += processed.slice(lastFinalIndex);
  if (!everInFinal) return '';
  return result.replace(FINAL_TAG_SCAN_RE, '');
}
console.log(stripWithPrRegex('<final reason="a>b">SECRET'));
NODE`
